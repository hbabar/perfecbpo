                } else {

                    mobileMenuIcon.classList.remove('fa-bars');

                    mobileMenuIcon.classList.add('fa-xmark');

                }

        // --- Calculator Logic ---

        const checkboxes = document.querySelectorAll('.calc-checkbox');

        const stageRadios = document.querySelectorAll('.stage-radio');

        const resultDisplay = document.getElementById('calc-result');

        const errorBox = document.getElementById('form-error');


        // Currency Formatter

        const formatter = new Intl.NumberFormat('en-US', {

            style: 'currency',

            currency: 'USD',

            maximumFractionDigits: 0, 

        });


        function calculateTotal() {

            let baseTotal = 0;

            

            // Add up selected services

            checkboxes.forEach(box => {

                if(box.checked) {

                    baseTotal += parseInt(box.value);

                }

            });


            // Get multiplier

            let multiplier = 0;

            let stageSelected = false;

            stageRadios.forEach(radio => {

                if(radio.checked) {

                    multiplier = parseFloat(radio.value);

                    stageSelected = true;

                }

            });


            // Calculate final

            const finalTotal = baseTotal * multiplier;


            // Hide error box if user makes a selection

            if(baseTotal > 0 && stageSelected) {

                errorBox.classList.add('hidden');

            }


            // Animate number change

            resultDisplay.style.opacity = 0;

            setTimeout(() => {

                if (baseTotal > 0 && !stageSelected) {

                    resultDisplay.textContent = "Select Stage";

                } else {

                    resultDisplay.textContent = formatter.format(finalTotal);

                }

                resultDisplay.style.opacity = 1;

            }, 150);

        }


        // Attach event listeners

        checkboxes.forEach(box => box.addEventListener('change', calculateTotal));

        stageRadios.forEach(radio => radio.addEventListener('change', calculateTotal));


        // Initial calculation on load

        calculateTotal();


        // --- Custom Error Helper (Avoids window alerts) ---

        function showFormError(message) {

            const errorTextSpan = errorBox.querySelector('span');

            if (errorTextSpan) {

                errorTextSpan.textContent = message;

            }

            errorBox.classList.remove('hidden');

            errorBox.scrollIntoView({ behavior: 'smooth', block: 'center' });

        }


        // --- Form Submission Handling ---

        async function handleLeadSubmit(e) {

            e.preventDefault();

            

            // Validate that at least one service is selected before submitting

            let serviceSelected = false;

            let selectedNames = [];

            checkboxes.forEach(box => { 

                if(box.checked) {

                    serviceSelected = true;

                    selectedNames.push(box.getAttribute('data-name'));

                }

            });

            

            if(!serviceSelected) {

                showFormError("Please select at least one service above to proceed.");

                return;

            }


            // Validate that business stage is selected

            let selectedStage = "";

            let stageSelected = false;

            stageRadios.forEach(radio => {

                if(radio.checked) {

                    stageSelected = true;

                    selectedStage = radio.parentNode.querySelector('span.font-bold').innerText.trim();

                }

            });


            if(!stageSelected) {

                showFormError("Please select your Business Stage to proceed.");

                return;

            }


            const form = document.getElementById('lead-form');

            const submitBtn = form.querySelector('button[type="submit"]');

            

            // Gather extra details for Formspree mapping

            const formData = new FormData(form);

            

            let finalCost = resultDisplay.textContent;


            // Map calculator state to form values

            formData.append('Estimated Monthly Investment', finalCost);

            formData.append('Selected Services', selectedNames.join(', '));

            formData.append('Business Scale', selectedStage);


            // Set loading state

            submitBtn.disabled = true;

            submitBtn.innerHTML = '<i class="fa-solid fa-spinner animate-spin mr-2"></i> Sending...';


            try {

                const response = await fetch("https://formspree.io/f/xzdwapwk", {

                    method: "POST",

                    body: formData,

                    headers: {

                        'Accept': 'application/json'

                    }

                });

                

                if(response.ok) {

                    // Hide form, hide error, show success message

                    const successMsg = document.getElementById('success-message');

                    errorBox.classList.add('hidden');

                    form.style.display = 'none';

                    successMsg.classList.remove('hidden');

                    successMsg.style.display = 'block';

                } else {

                    const data = await response.json();

                    const errMsg = data.errors ? data.errors.map(err => err.message).join(", ") : "There was a problem submitting your form. Please try again.";

                    showFormError(errMsg);

                    submitBtn.disabled = false;

                    submitBtn.innerHTML = 'Send My Custom Proposal <i class="fa-solid fa-paper-plane text-sm"></i>';

                }

            } catch(err) {

                showFormError("Network error. Please check your connection and try again.");

                submitBtn.disabled = false;

                submitBtn.innerHTML = 'Send My Custom Proposal <i class="fa-solid fa-paper-plane text-sm"></i>';

            }

        }

    </script>
